import numpy as np
from numpy import linalg as LA

__all__ = ['unitA', 'closeV', 'parallel', 'ptOrtho', 'findClosest']



[docs]def unitA(array):
    """ Returns the unit array """
    return array / LA.norm(array)



def angleA(a1, a2):
    """ Returns the angle in radians between arrays 'a1' and 'a2' """
    a1_u = unitA(a1)
    a2_u = unitA(a2)
    return np.arccos(np.clip(np.dot(a1_u, a2_u), -1.0, 1.0))



[docs]def closeV(iv, V, radius, displ=None):
    """ Returns an array off 1st neighbours indexes """
    idx = np.empty(shape=[0], dtype=np.int32)
    Vd = V.copy()
    if np.any(displ):
        Vd = Vd + displ
    for i in range(iv):
        if LA.norm(Vd[i]-V[iv]) < radius:
            idx = np.append(idx, [[i]])
    for i in range(iv+1, len(V)):
        if LA.norm(Vd[i]-V[iv]) < radius:
            idx = np.append(idx, [[i]])
    return idx




[docs]def parallel(pt1, pt2):
    """
    Returns a pair of points in :math:`\mathbb{R}^2` belonging to
    a line parallel to the one defined by the provided points.
    """
    if pt2[1] >= pt1[1]:
        v12 = pt2 - pt1
        theta = angleA(v12, [1., 0., 0.])
        parpt1 = (pt1[0] + 0.2*np.cos(theta), pt1[1] + 0.2*np.sin(theta))
        parpt2 = (pt2[0] - 0.2*np.cos(theta), pt2[1] - 0.2*np.sin(theta))
    else:
        v12 = pt1 - pt2
        theta = angleA(v12, [1., 0., 0.])
        parpt1 = (pt1[0] - 0.2*np.cos(theta), pt1[1] - 0.2*np.sin(theta))
        parpt2 = (pt2[0] + 0.2*np.cos(theta), pt2[1] + 0.2*np.sin(theta))
    theta = theta - np.pi/2
    parpt1 = (parpt1[0] - 0.25*np.cos(theta), parpt1[1] - 0.25*np.sin(theta))
    parpt2 = (parpt2[0] - 0.25*np.cos(theta), parpt2[1] - 0.25*np.sin(theta))
    return parpt1, parpt2




[docs]def ptOrtho(v1, v2, v3):
    """
    Return a point slightly distant from 'v2' and belonging to the
    line orthogonal to (:math:`\overline{v2-v1}+\overline{v3-v2}`)
    and that contains 'v2'.
    """
    v12 = v2 - v1
    v23 = v3 - v2
    theta = angleA(v12, v23)
    ortho = (v12 - v23) * np.sin(0.5*theta)
    ortho = unitA(ortho)
    return (v2 + 0.35*ortho)




[docs]def findClosest(array, value):
    """
    Returns the element index from 'array'
    that is closer to 'value'.
    """
    array = np.asarray(array)
    idx = np.searchsorted(array, value)
    return idx
